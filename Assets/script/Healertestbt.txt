tree("Root")
    repeat
        fallback
            tree("ReturnPokebot")
            tree("DeliverPotion")
            tree("CreatePotion")
            tree("FindIngredients")
            tree("IdentifyWound")
            tree("Approach")
            tree("Idle")

tree("Idle")
    sequence
        Display("No customers, Time to Idle")
        Idle

tree("Approach")
    sequence
        Approach
        fallback
            IsInteract
            sequence
                QueryPlayer("hael smt")
                fallback
                    sequence
                        IsKeyDown("Y")
                        Display("Please follow me.")
                        StartInteract
                    sequence
                        IsKeyDown("N")
                        Display("Please enjoy the exhibits.")

tree("IdentifyWound")
    sequence
        IsInteract
        sequence
            IdentifyWound

tree("FindIngredients")
    sequence
        IsPokebotSeated
        sequence
            FindIngredients

tree("CreatePotion")
    sequence
        AllIngredientsFound
        sequence
            CreatePotion

tree("DeliverPotion")
    sequence
        IsPotionCreated
        sequence
            DeliverPotion

tree("ReturnPokebot")
    sequence
        IsPotionDelivered
        sequence
            ReturnPokebot
